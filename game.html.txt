    <div id="threat-indicator">Threat Level: Normal</div>
    <script>
        let waveCount = 0;
        let bossWaveInterval = 10;
        let threatLevel = "Normal";

        function startGame() {
            // Initialize game and smarter AI
            spawnEnemies();
            updateThreatIndicator();
        }

        function spawnEnemies() {
            // Logic to spawn enemies intelligently
            waveCount++;
            if (waveCount % bossWaveInterval === 0) {
                spawnBoss();
            }
            // Other enemy spawning logic
            updateThreatIndicator();
        }

        function spawnBoss() {
            // Logic for boss enemy
            alert("A boss has appeared!");
            threatLevel = "High!";
        }

        function updateThreatIndicator() {
            document.getElementById('threat-indicator').innerText = `Threat Level: ${threatLevel}`;
        }

        startGame();
    </script>
