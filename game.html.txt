// --- AUDIO SYSTEM (Procedural 8-bit Sounds) ---
let audioCtx = null;
let soundEnabled = true;

function initAudio() {
    if (!audioCtx) {
        const AudioContext = window.AudioContext || window.webkitAudioContext;
        audioCtx = new AudioContext();
        startBGM(); // Start background music scheduler
    }
    if (audioCtx.state === 'suspended') {
        audioCtx.resume();
    }
}

function playTone(frequency, type, duration, volume = 0.05) {
    if (!audioCtx) return;
    const osc = audioCtx.createOscillator();
    const gainNode = audioCtx.createGain();
    
    osc.type = type;
    osc.frequency.setValueAtTime(frequency, audioCtx.currentTime);
    if (type === 'square' || type === 'sawtooth') {
        osc.frequency.exponentialRampToValueAtTime(frequency * 0.5, audioCtx.currentTime + duration);
    }
    
    gainNode.gain.setValueAtTime(volume, audioCtx.currentTime);
    gainNode.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + duration);
    
    osc.connect(gainNode);
    gainNode.connect(audioCtx.destination);
    
    osc.start();
    osc.stop(audioCtx.currentTime + duration);
}

function playNoise(duration, volume = 0.1, lowpassFreq = 1000) {
    if (!audioCtx) return;
    const bufferSize = audioCtx.sampleRate * duration;
    const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
    const data = buffer.getChannelData(0);
    for (let i = 0; i < bufferSize; i++) {
        data[i] = Math.random() * 2 - 1;
    }
    const noiseSource = audioCtx.createBufferSource();
    noiseSource.buffer = buffer;
    
    const filter = audioCtx.createBiquadFilter();
    filter.type = 'lowpass';
    filter.frequency.value = lowpassFreq;
    
    const gainNode = audioCtx.createGain();
    gainNode.gain.setValueAtTime(volume, audioCtx.currentTime);
    gainNode.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + duration);
    
    noiseSource.connect(filter);
    filter.connect(gainNode);
    gainNode.connect(audioCtx.destination);
    
    noiseSource.start();
}

// DEBOUNCER: Prevent massive overlapping sounds of the same type
const lastPlayed = {};
function throttleSound(key, playFn) {
    if (!soundEnabled || !audioCtx) return;
    const now = audioCtx.currentTime;
    // Only play if this specific sound hasn't played in the last 50ms
    if (!lastPlayed[key] || now - lastPlayed[key] > 0.05) { 
        playFn();
        lastPlayed[key] = now;
    }
}

// --- BACKGROUND MUSIC ENGINE ---
let bgmTrack = 'menu';
let bgmStep = 0;
let nextBgmTime = 0;
let bgmTimerID = null;

const bgmTempo = { menu: 100, selection: 140, play: 160 };
const bgmNotes = {
    menu: [ 130.81, null, 196.00, null, 155.56, null, 196.00, null, 130.81, null, 196.00, null, 233.08, null, 196.00, null ], // Mysterious
    selection: [ 261.63, null, 261.63, null, 311.13, null, null, null, 196.00, null, 196.00, null, 233.08, null, null, null ], // Upbeat blips
    play: [ 65.41, 65.41, null, 77.78, null, 87.31, null, 65.41, 65.41, 65.41, null, 98.00, null, 116.54, null, 65.41 ] // Driving bassline
};

function bgmPlayNote(freq, type, vol, time) {
    if (!audioCtx) return;
    const osc = audioCtx.createOscillator();
    const gain = audioCtx.createGain();
    osc.frequency.value = freq;
    osc.type = type;
    gain.gain.setValueAtTime(0, time);
    gain.gain.linearRampToValueAtTime(vol, time + 0.02);
    gain.gain.exponentialRampToValueAtTime(0.001, time + 0.15);
    osc.connect(gain);
    gain.connect(audioCtx.destination);
    osc.start(time);
    osc.stop(time + 0.15);
}

function bgmScheduler() {
    if (!audioCtx) return;
    while (nextBgmTime < audioCtx.currentTime + 0.1) {
        if (soundEnabled && bgmTrack !== 'none' && bgmNotes[bgmTrack]) {
            let freq = bgmNotes[bgmTrack][bgmStep];
            if (freq) {
                let type = bgmTrack === 'play' ? 'sawtooth' : (bgmTrack === 'selection' ? 'square' : 'sine');
                let vol = bgmTrack === 'play' ? 0.03 : 0.02;
                bgmPlayNote(freq, type, vol, nextBgmTime);
            }
        }
        
        let tempo = bgmTempo[bgmTrack] || 120;
        let secondsPerBeat = 60.0 / tempo;
        nextBgmTime += 0.25 * secondsPerBeat; // 16th notes
        bgmStep = (bgmStep + 1) % 16;
    }
}

function startBGM() {
    if (bgmTimerID) return; 
    nextBgmTime = audioCtx.currentTime + 0.1;
    bgmTimerID = setInterval(bgmScheduler, 25);
}

class Cell {
    constructor(x, y) {
        this.x = x; this.y = y; this.width = CELL_SIZE; this.height = CELL_SIZE;
    }

    draw() {
        ctx.strokeStyle = '#333'; ctx.lineWidth = 1;
        ctx.strokeRect(this.x, this.y, this.width, this.height);

        // Hover effects
        if (gameMode === 'defend' && gameActive && !isPaused &&
            mouse.x >= this.x && mouse.x < this.x + this.width &&
            mouse.y >= this.y && mouse.y < this.y + this.height) {
            
            let occupied = defenders.some(d => d.x === this.x && d.y === this.y);

            if (shovelActive && occupied) {
                ctx.fillStyle = 'rgba(255, 0, 0, 0.4)';
                ctx.fillRect(this.x, this.y, this.width, this.height);
                ctx.font = '40px Arial'; ctx.fillStyle = 'white'; ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
                ctx.fillText("🗑️", this.x + this.width/2, this.y + this.height/2);
            } else if (selectedUnit) {
                ctx.fillStyle = occupied ? 'rgba(255,50,50,0.4)' : 'rgba(255,255,255,0.2)';
                ctx.fillRect(this.x, this.y, this.width, this.height);
            }
        }

        if (gameMode === 'attack' && gameActive && selectedUnit && !isPaused) {
            if (mouse.y >= this.y && mouse.y < this.y + this.height && mouse.x > LEFT_OFFSET && mouse.x < LOGICAL_WIDTH) {
                ctx.fillStyle = 'rgba(0, 255, 100, 0.05)';
                ctx.fillRect(LEFT_OFFSET, this.y, GRID_COLS * CELL_SIZE, this.height);
                ctx.fillStyle = 'rgba(0, 255, 100, 0.3)';
                ctx.fillRect(LOGICAL_WIDTH - 60, this.y, 60, this.height);
            }
        }
    }
}

class Defender {
    constructor(x, y, type) {
        this.x = x; this.y = y; this.width = CELL_SIZE; this.height = CELL_SIZE;
        this.type = type;
        this.hp = type.hp; this.maxHp = type.hp;
        this.timer = 0;
        this.shootingTimer = 0;
        
        if (this.type.type === 'instant') { this.life = 0; }
        if (this.type.type === 'explosive') { this.timer = 0; }
    }

    draw() {
        ctx.globalAlpha = 1.0;
        let bounce = Math.sin(frames * 0.08) * 3;
        
        if (this.type.type === 'trap') {
             ctx.fillStyle = 'rgba(255, 200, 0, 0.3)';
             ctx.fillRect(this.x + 5, this.y + 5, this.width - 10, this.height - 10);
             ctx.font = '40px ' + GAME_FONT;
             bounce = 0;
        } else if (this.type.type === 'explosive') {
             let scale = 1 + Math.sin(frames * 0.5) * 0.1;
             ctx.fillStyle = 'rgba(255,0,0,0.3)'; 
             ctx.beginPath();
             ctx.arc(this.x + CELL_SIZE/2, this.y + CELL_SIZE/2, 30 * scale, 0, Math.PI*2);
             ctx.fill();
             ctx.font = '60px ' + GAME_FONT;
             bounce = Math.random() * 5; 
        } else {
             ctx.fillStyle = 'rgba(0,0,0,0.6)'; ctx.beginPath();
             ctx.ellipse(this.x + CELL_SIZE/2, this.y + CELL_SIZE - 10, 28, 10, 0, 0, Math.PI*2); ctx.fill();
             
             ctx.fillStyle = 'rgba(255, 255, 255, 0.15)';
             ctx.beginPath();
             ctx.arc(this.x + CELL_SIZE/2, this.y + CELL_SIZE/2 + bounce - 5, 30, 0, Math.PI * 2);
             ctx.fill();
             
             ctx.font = '60px ' + GAME_FONT;
        }

        ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
        if (this.type.type === 'healer') bounce = Math.sin(frames * 0.05) * 5; 

        ctx.fillStyle = 'rgba(0,0,0,0.5)';
        ctx.fillText(this.type.emoji, this.x + CELL_SIZE/2, this.y + CELL_SIZE/2 + bounce - 5 + 4);

        ctx.fillStyle = '#ffffff'; 
        ctx.fillText(this.type.emoji, this.x + CELL_SIZE/2, this.y + CELL_SIZE/2 + bounce - 5);
        ctx.globalAlpha = 1.0;

        if (this.type.type === 'healer' && this.timer % this.type.cooldown > this.type.cooldown - 15) {
             ctx.strokeStyle = 'rgba(0, 255, 100, 0.5)'; ctx.lineWidth = 3;
             ctx.beginPath();
             ctx.arc(this.x + CELL_SIZE/2, this.y + CELL_SIZE/2, CELL_SIZE * 1.5 * ((this.timer % this.type.cooldown) / this.type.cooldown), 0, Math.PI*2);
             ctx.stroke();
        }

        if (this.hp < this.maxHp && this.type.type !== 'instant' && this.type.type !== 'explosive') {
            ctx.fillStyle = '#ff0000'; ctx.fillRect(this.x + 10, this.y + 5, CELL_SIZE - 20, 6);
            ctx.fillStyle = '#00ff00'; ctx.fillRect(this.x + 10, this.y + 5, (CELL_SIZE - 20) * (this.hp / this.maxHp), 6);
        }
    }

    update() {
        this.timer++;

        if (this.type.trait === 'regen' && frames % 60 === 0 && this.hp < this.maxHp) this.hp += 10;

        if (this.type.type === 'producer') {
            if (this.timer % this.type.cooldown === 0) {
                if (gameMode === 'defend') {
                    resources.push(new Resource(this.x + CELL_SIZE/2, this.y + CELL_SIZE/2));
                    addFloatyText("POP!", this.x + CELL_SIZE/2, this.y);
                } else {
                    aiResources += 20;
                    addFloatyText("⚡", this.x + CELL_SIZE/2, this.y);
                }
            }
        }
        
        if (this.type.type === 'healer') {
            if (this.timer > 0 && this.timer % this.type.cooldown === 0) {
                let healed = false;
                defenders.forEach(d => {
                    if (d !== this && d.hp < d.maxHp && Math.hypot(d.x - this.x, d.y - this.y) < CELL_SIZE * 1.6) {
                        d.hp = Math.min(d.maxHp, d.hp + this.type.healAmount);
                        addFloatyText("💚", d.x + CELL_SIZE/2, d.y);
                        healed = true;
                    }
                });
                if (healed) addFloatyText("HEAL!", this.x + CELL_SIZE/2, this.y - 10);
            }
        }

        if (this.type.type === 'shooter' || this.type.type === 'lobber') {
            this.shootingTimer++;
            if (this.shootingTimer >= this.type.fireRate) {
                let laneY = this.y;
                let enemyInLane = enemies.some(e => e.y === laneY && e.x > this.x && e.x < this.x + (this.type.range * CELL_SIZE));

                if (enemyInLane || this.type.pattern === 'tri') {
                    if (this.type.type === 'lobber') sfx.lob();
                    else sfx.shoot();

                    if (enemyInLane) projectiles.push(new Projectile(this.x + CELL_SIZE/2, this.y + CELL_SIZE/2, this.type.projectileColor, this.type.damage, 0, 0, this.type.effect, this.type.type === 'lobber'));
                    
                    if (this.type.pattern === 'tri' && enemyInLane) {
                        projectiles.push(new Projectile(this.x + CELL_SIZE/2, this.y + CELL_SIZE/2, this.type.projectileColor, this.type.damage, 0, -1.5, this.type.effect));
                        projectiles.push(new Projectile(this.x + CELL_SIZE/2, this.y + CELL_SIZE/2, this.type.projectileColor, this.type.damage, 0, 1.5, this.type.effect));
                    }
                    this.shootingTimer = 0;
                }
            }
        }

        if (this.type.type === 'melee') {
            this.shootingTimer++;
            if (this.shootingTimer >= this.type.fireRate) {
                 let target = enemies.find(e => e.y === this.y && e.x > this.x && e.x < this.x + (this.type.range * CELL_SIZE));
                 if (target) {
                     target.takeDamage(this.type.damage);
                     sfx.melee();
                     
                     if (this.type.effect === 'knockback') {
                         target.x += 45; 
                         target.status.stun = 15; 
                         addFloatyText("WHACK!", this.x + CELL_SIZE, this.y);
                     } else if (this.type.effect === 'splash_entangle') {
                         enemies.forEach(ne => {
                             if (ne !== target && Math.hypot(ne.x - target.x, ne.y - target.y) < CELL_SIZE * 1.5) {
                                 ne.takeDamage(this.type.damage * 0.5);
                                 ne.applyEffect('slow');
                             }
                         });
                         target.applyEffect('slow');
                         addFloatyText("SPLAT!", this.x + CELL_SIZE, this.y);
                     } else {
                         addFloatyText("CHOP!", this.x + CELL_SIZE, this.y);
                         if (this.type.effect === 'entangle') target.applyEffect('slow');
                     }

                     if (this.type.id === 'baguette') {
                         this.hp -= 15; 
                     }
                     
                     this.shootingTimer = 0;
                 }
            }
        }

        if (this.type.type === 'explosive') {
            if (this.timer > 40) { 
                enemies.forEach(e => {
                    let dist = Math.hypot(e.x - (this.x + CELL_SIZE/2), e.y - (this.y + CELL_SIZE/2));
                    if (dist < CELL_SIZE * 1.8) e.takeDamage(this.type.damage);
                });
                addFloatyText("KABOOM!", this.x, this.y);
                sfx.boom();
                this.hp = 0;
            }
        }

        if (this.type.type === 'instant') {
            this.life++;
            if (this.life > 100) this.hp = 0;
        }
    }
}

class Enemy {
    constructor(y, type, difficultyWave = 1) {
        this.x = LOGICAL_WIDTH; this.y = y; this.width = CELL_SIZE; this.height = CELL_SIZE;
        this.type = type;
        
        let hpMult = gameMode === 'defend' ? (1 + (difficultyWave * 0.3)) : 1;
        this.hp = type.hp * hpMult; 
        this.maxHp = this.hp;
        
        let speedMult = gameMode === 'defend' ? (1 + (difficultyWave * 0.05)) : 1;
        this.baseSpeed = Math.min(type.speed * speedMult, 3.5);
        this.speed = this.baseSpeed;
        this.movement = this.speed;
        
        this.eating = false;
        this.flashTimer = 0;
        this.status = { slow: 0, stun: 0, burn: 0 };
    }

    update() {
        if (this.flashTimer > 0) this.flashTimer--;
        if (this.type.trait === 'regen' && this.flashTimer <= 0 && frames % 60 === 0 && this.hp < this.maxHp) this.hp += 5; 

        if (this.type.emoji === '🎃' && frames % 180 === 0 && this.x < LOGICAL_WIDTH - 50) {
             let minionType = Math.random() > 0.5 ? ENEMIES.CARROT : ENEMIES.TOMATO;
             let minion = new Enemy(this.y, minionType, wave);
             minion.x = this.x + CELL_SIZE; 
             enemies.push(minion);
             addFloatyText("SPAWN!", this.x, this.y - 40);
        }

        if (this.status.burn > 0) {
            this.hp -= 0.5;
            this.status.burn--;
            if (this.status.burn % 20 === 0) addFloatyText("🔥", this.x + CELL_SIZE/2, this.y + 10);
        }

        if (this.status.stun > 0) {
            this.status.stun--; this.speed = 0;
        } else if (this.status.slow > 0) {
            this.speed = this.baseSpeed * 0.5; this.status.slow--;
        } else {
            this.speed = this.baseSpeed;
        }

        this.eating = false;
        let collision = false;
        
        for (let i = 0; i < defenders.length; i++) {
            let d = defenders[i];
            if (this.x < d.x + d.width - 25 && this.x + this.width - 25 > d.x && this.y === d.y) {
                if (d.type.type === 'trap') {
                    if (frames % 20 === 0) { d.hp -= 5; this.takeDamage(d.type.damage); }
                    this.speed = this.baseSpeed * 0.5; 
                    continue; 
                }

                if (d.type.type === 'explosive') {
                    this.eating = true; this.movement = 0; collision = true;
                } else if (d.type.type === 'instant') {
                    d.hp = 0; this.takeDamage(d.type.damage); addFloatyText("SQUASH!", d.x, d.y); sfx.squash();
                } else {
                    this.eating = true; this.movement = 0; d.hp -= this.type.damage; collision = true;
                }
                break;
            }
        }

        if (!collision && this.status.stun <= 0) {
            this.movement = this.speed; this.x -= this.movement;
        }

        if (this.x < 0) {
            if (gameMode === 'defend') {
                endGame(false);
            } else {
                endGame(true);
            }
        }
    }

    draw() {
        ctx.globalAlpha = 1.0;
        let emojiX = this.x + CELL_SIZE/2;
        let hop = this.eating ? Math.sin(frames * 0.4) * 2 : Math.abs(Math.sin(frames * 0.1)) * 10;
        let emojiY = this.y + CELL_SIZE/2 - hop - 5;
        
        ctx.fillStyle = 'rgba(0,0,0,0.6)'; ctx.beginPath();
        ctx.ellipse(emojiX, this.y + CELL_SIZE - 5, 25, 8, 0, 0, Math.PI*2); ctx.fill();

        ctx.fillStyle = gameMode === 'defend' ? 'rgba(255, 50, 50, 0.15)' : 'rgba(50, 255, 50, 0.15)';
        ctx.beginPath();
        ctx.arc(emojiX, emojiY, 30, 0, Math.PI * 2);
        ctx.fill();

        ctx.font = '65px ' + GAME_FONT; 
        ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
        
        ctx.fillStyle = 'rgba(0,0,0,0.5)';
        ctx.fillText(this.type.emoji, emojiX, emojiY + 4);

        ctx.fillStyle = '#ffffff';
        ctx.fillText(this.type.emoji, emojiX, emojiY);
        
        if (this.status.stun > 0) {
            ctx.fillStyle = 'rgba(255, 255, 0, 0.3)';
            ctx.beginPath(); ctx.arc(emojiX, emojiY, 35, 0, Math.PI*2); ctx.fill();
        }
        
        if (this.status.slow > 0) {
            ctx.fillStyle = '#ffffff';
            ctx.fillText("❄️", emojiX + 25, this.y + 20);
        }

        if (this.flashTimer > 0) {
            ctx.fillStyle = 'rgba(255, 255, 255, 0.6)';
            ctx.beginPath(); ctx.arc(emojiX, emojiY, 30, 0, Math.PI*2); ctx.fill();
        }

        if (this.hp < this.maxHp) {
            ctx.fillStyle = '#ff0000'; ctx.fillRect(this.x + 15, this.y, CELL_SIZE - 30, 6);
            ctx.fillStyle = '#00ff00'; ctx.fillRect(this.x + 15, this.y, (CELL_SIZE - 30) * (this.hp / this.maxHp), 6);
        }
    }

    takeDamage(amount) { 
        this.hp -= amount; 
        this.flashTimer = 5; 
        if (Math.random() > 0.5) sfx.hit(); 
    }
    applyEffect(effect) {
        if (effect === 'slow') this.status.slow = 120;
        if (effect === 'stun') this.status.stun = 20; // Reduced stun time as requested
        if (effect === 'burn') this.status.burn = 120;
    }
}

class Projectile {
    constructor(x, y, color, damage, vx = 0, vy = 0, effect = null, isLobbed = false) {
        this.x = x; this.y = y; this.width = 12; this.height = 12;
        this.speed = 6; this.color = color; this.damage = damage;
        this.vx = vx; this.vy = vy; this.effect = effect;
        this.isLobbed = isLobbed;
        this.hitList = []; 
        this.markedForDeletion = false;
        this.lifeTime = 0;
    }

    update() {
        this.lifeTime++;
        
        if (this.effect === 'boomerang') {
            if (this.lifeTime === 60) {
                this.speed = -this.speed; 
                this.hitList = []; 
            }
        }

        this.x += this.speed + this.vx; this.y += this.vy;
        
        if (this.x > LOGICAL_WIDTH || this.x < -20 || this.y < 0 || this.y > LOGICAL_HEIGHT) this.markedForDeletion = true;

        for (let i = 0; i < enemies.length; i++) {
            let e = enemies[i];
            if (this.x > e.x && this.x < e.x + CELL_SIZE && this.y > e.y && this.y < e.y + CELL_SIZE) {
                if (this.effect === 'pierce' || this.effect === 'boomerang') {
                    if (!this.hitList.includes(e)) { 
                        e.takeDamage(this.damage); 
                        if (this.effect) e.applyEffect(this.effect);
                        this.hitList.push(e); 
                    }
                } else if (this.effect === 'splash') {
                    e.takeDamage(this.damage);
                    enemies.forEach(ne => { if (Math.hypot(ne.x - this.x, ne.y - this.y) < 100) ne.takeDamage(this.damage / 2); });
                    this.markedForDeletion = true; addFloatyText("POW!", this.x, this.y); sfx.boom(); break;
                } else {
                    e.takeDamage(this.damage); if (this.effect) e.applyEffect(this.effect);
                    this.markedForDeletion = true; break;
                }
            }
        }
    }

    draw() {
        ctx.fillStyle = this.color;
        
        if (this.effect === 'boomerang') {
             ctx.save();
             ctx.translate(this.x, this.y);
             ctx.rotate(this.lifeTime * 0.2); 
             ctx.globalAlpha = 0.4;
             ctx.beginPath(); ctx.arc(0, 0, 15, 0, Math.PI, true); ctx.fill();
             ctx.globalAlpha = 1.0;
             ctx.beginPath();
             ctx.arc(0, 0, 10, 0, Math.PI, true); 
             ctx.closePath();
             ctx.fill();
             ctx.restore();
        } else {
             ctx.globalAlpha = 0.4;
             ctx.beginPath(); ctx.arc(this.x, this.y, 13, 0, Math.PI * 2); ctx.fill();
             ctx.globalAlpha = 1.0;
             ctx.beginPath(); ctx.arc(this.x, this.y, 8, 0, Math.PI * 2); ctx.fill();
             ctx.fillStyle = 'rgba(255,255,255,0.8)'; 
             ctx.beginPath(); ctx.arc(this.x - 2, this.y - 2, 3, 0, Math.PI * 2); ctx.fill();
        }
    }
}

class Resource {
    constructor(x, y) {
        this.x = x; this.y = y; this.width = 60; this.height = 60; this.amount = 25;
        this.markedForDeletion = false; this.lifeTime = 0; this.rotation = 0;
        this.targetY = y;
        if (y < 0) {
            this.x = Math.random() * (LOGICAL_WIDTH - 100) + 50; this.y = -50;
            this.targetY = Math.random() * (LOGICAL_HEIGHT - 200) + 100;
        }
    }

    update() {
        this.lifeTime++; 
        this.rotation = Math.sin(this.lifeTime * 0.05) * 0.2;
        if (this.y < this.targetY) this.y += 3;
        
        if (this.lifeTime > 300) this.markedForDeletion = true; 
        
        if (mouse.x >= this.x - 30 && mouse.x <= this.x + 30 &&
            mouse.y >= this.y - 30 && mouse.y <= this.y + 30) {
            resourceAmount += this.amount; this.markedForDeletion = true;
            addFloatyText("+" + this.amount, this.x, this.y); updateCardVisuals();
            sfx.coin();
        }
    }

    draw() {
        ctx.save(); 
        ctx.translate(this.x, this.y); 
        ctx.rotate(this.rotation);
        ctx.fillStyle = gameMode === 'defend' ? 'rgba(255, 255, 0, 0.25)' : 'rgba(0, 255, 255, 0.25)';
        ctx.beginPath();
        ctx.arc(0, 0, 25, 0, Math.PI * 2);
        ctx.fill();
        ctx.font = '45px ' + GAME_FONT; 
        ctx.textAlign = 'center'; 
        ctx.textBaseline = 'middle';
        ctx.fillStyle = 'rgba(0,0,0,0.5)';
        ctx.fillText(resourceIcon, 0, 4);
        ctx.fillStyle = '#ffffff';
        ctx.fillText(resourceIcon, 0, 0); 
        ctx.restore();
    }
}

class FloatyText {
    constructor(text, x, y) { this.text = text; this.x = x; this.y = y; this.timer = 0; this.markedForDeletion = false; }
    update() { this.y -= 1.5; this.timer++; if (this.timer > 40) this.markedForDeletion = true; }
    draw() {
        ctx.font = 'bold 24px ' + GAME_FONT; ctx.fillStyle = '#ffff00'; ctx.strokeStyle = 'black';
        ctx.lineWidth = 3; ctx.strokeText(this.text, this.x, this.y); ctx.fillText(this.text, this.x, this.y);
    }
}

// --- GAME LOOP ---

function setupWave() {
    enemiesToSpawn = [];
    let difficulty = wave === 1 ? 8 : Math.floor(wave * 8 + 5 + Math.pow(wave, 2.2)); 
    
    for (let i = 0; i < Math.floor(difficulty / 2); i++) enemiesToSpawn.push('CARROT');
    if (wave > 1) for (let i = 0; i < Math.floor(wave / 2); i++) enemiesToSpawn.push('TOMATO');
    if (wave > 2) for (let i = 0; i < Math.floor(wave / 2); i++) enemiesToSpawn.push('CORN');
    if (wave > 3) for (let i = 0; i < Math.floor(wave / 3); i++) enemiesToSpawn.push('ONION');
    if (wave > 4) for (let i = 0; i < Math.floor(wave / 3); i++) enemiesToSpawn.push('CHILI');
    
    if (wave > 6) for (let i = 0; i < Math.floor(wave / 4); i++) enemiesToSpawn.push('EGGPLANT');
    if (wave > 7) for (let i = 0; i < Math.floor(wave / 2); i++) enemiesToSpawn.push('GARLIC');
    if (wave > 8) for (let i = 0; i < Math.floor(wave / 4); i++) enemiesToSpawn.push('SPINACH');
    
    if (wave % 5 === 0 && wave > 4) enemiesToSpawn.push('EGGPLANT');
    if (wave % 10 === 0) enemiesToSpawn.push('PUMPKIN');

    enemiesToSpawn.sort(() => Math.random() - 0.5);
    showMessage(`WAVE ${wave}`);
    waveStarted = true;
}

function handleSpawns() {
    let dropRate = gameMode === 'defend' ? 600 : 300; 
    if (frames % dropRate === 0 && frames > 0) resources.push(new Resource(-100, -100));
    
    if (gameMode === 'defend' && enemiesToSpawn.length > 0) {
        let spawnRate = wave === 1 ? 240 : Math.max(20, 180 - (wave * 20)); 
        if (frames % spawnRate === 0) {
            let type = enemiesToSpawn.pop();
            let row = Math.floor(Math.random() * GRID_ROWS);
            enemies.push(new Enemy(TOP_OFFSET + (row * CELL_SIZE), ENEMIES[type], wave));
        }
    }
}

function handleAI() {
    if (frames % 60 === 0) {
        aiResources += 15 + Math.floor(frames / 300); 
        attackSurvivalTime++;
    }

    if (aiPlacementCooldown > 0) {
        aiPlacementCooldown--;
        return;
    }

    let myProducers = defenders.filter(d => d.type.type === 'producer').length;

    let laneThreats = [0, 0, 0, 0, 0];
    let laneAdvancedX = [LOGICAL_WIDTH, LOGICAL_WIDTH, LOGICAL_WIDTH, LOGICAL_WIDTH, LOGICAL_WIDTH];
    
    enemies.forEach(e => {
        let row = Math.floor((e.y - TOP_OFFSET) / CELL_SIZE);
        if(row >= 0 && row < GRID_ROWS) {
            laneThreats[row] += e.hp; 
            if (e.x < laneAdvancedX[row]) {
                laneAdvancedX[row] = e.x;
            }
        }
    });

    let targetColMin = 0;
    let targetColMax = 8;
    let targetRow = -1;
    let unitToPlace = null;

    let maxThreat = Math.max(...laneThreats);
    let wantsProducer = (myProducers < 6 && Math.random() < 0.7) || (myProducers < 3);

    if (maxThreat > 0) {
        let threatenedLanes = [];
        laneThreats.forEach((threat, index) => {
            if (threat >= maxThreat * 0.8 && threat > 0) threatenedLanes.push(index);
        });
        targetRow = threatenedLanes[Math.floor(Math.random() * threatenedLanes.length)];
    } else {
        let rowDefenders = [0, 0, 0, 0, 0];
        defenders.forEach(d => {
            let row = Math.floor((d.y - TOP_OFFSET) / CELL_SIZE);
            if(row >= 0 && row < GRID_ROWS) rowDefenders[row]++;
        });
        let minDef = Math.min(...rowDefenders);
        let weakRows = [];
        rowDefenders.forEach((count, index) => {
            if (count === minDef) weakRows.push(index);
        });
        targetRow = weakRows[Math.floor(Math.random() * weakRows.length)];
    }

    if (wantsProducer && maxThreat === 0) {
        let prodDeck = aiDeck.filter(u => u.type === 'producer');
        if (prodDeck.length > 0) {
            unitToPlace = prodDeck[Math.floor(Math.random() * prodDeck.length)];
            targetColMin = 0; targetColMax = 2;
        }
    }
    else if (maxThreat > 0 && Math.random() < 0.9) {
        let blockerDeck = aiDeck.filter(u => ['wall', 'melee', 'trap', 'explosive'].includes(u.type));
        let shooterDeck = aiDeck.filter(u => ['shooter', 'lobber', 'healer'].includes(u.type));

        if (blockerDeck.length > 0 && Math.random() < 0.6) {
            unitToPlace = blockerDeck[Math.floor(Math.random() * blockerDeck.length)];
            targetColMin = 4; targetColMax = 8;
        } else if (shooterDeck.length > 0) {
            unitToPlace = shooterDeck[Math.floor(Math.random() * shooterDeck.length)];
            targetColMin = 1; targetColMax = 5;
        }
    }

    if (!unitToPlace) {
        unitToPlace = aiDeck[Math.floor(Math.random() * aiDeck.length)];
        if (unitToPlace.type === 'producer') { targetColMin = 0; targetColMax = 2; }
        else if (['shooter', 'lobber', 'healer'].includes(unitToPlace.type)) { targetColMin = 1; targetColMax = 5; }
        else { targetColMin = 4; targetColMax = 8; }
    }

    if (aiResources < unitToPlace.cost) {
        if (maxThreat > 0) {
            let affordable = aiDeck.filter(u => u.cost <= aiResources && u.type !== 'producer');
            if (affordable.length > 0) {
                unitToPlace = affordable[Math.floor(Math.random() * affordable.length)];
                if (['shooter', 'lobber', 'healer'].includes(unitToPlace.type)) { targetColMin = 1; targetColMax = 5; }
                else { targetColMin = 4; targetColMax = 8; }
            } else {
                aiPlacementCooldown = 15;
                return;
            }
        } else {
            aiPlacementCooldown = 15;
            return;
        }
    }

    const getValidCells = (strictColumn, strictRow, strictFront) => {
        return grid.filter(c => {
            if (defenders.some(d => d.x === c.x && d.y === c.y)) return false; 
            
            let col = Math.floor((c.x - LEFT_OFFSET) / CELL_SIZE);
            let row = Math.floor((c.y - TOP_OFFSET) / CELL_SIZE);
            
            if (strictColumn && (col < targetColMin || col > targetColMax)) return false;
            if (strictRow && targetRow !== -1 && row !== targetRow) return false;
            if (strictFront && row >= 0 && row < GRID_ROWS && c.x > laneAdvancedX[row] - (CELL_SIZE / 2)) return false;
            
            return true;
        });
    };

    let validCells = getValidCells(true, true, true);
    if (validCells.length === 0) validCells = getValidCells(false, true, true);
    if (validCells.length === 0) validCells = getValidCells(false, false, true);
    if (validCells.length === 0) validCells = getValidCells(false, false, false);

    if (validCells.length > 0) {
        if (unitToPlace.type === 'wall' || unitToPlace.type === 'melee') {
             validCells.sort((a,b) => b.x - a.x);
             let bestSpots = validCells.slice(0, 3);
             let cell = bestSpots[Math.floor(Math.random() * bestSpots.length)];
             defenders.push(new Defender(cell.x, cell.y, unitToPlace));
        } else {
             let cell = validCells[Math.floor(Math.random() * validCells.length)];
             defenders.push(new Defender(cell.x, cell.y, unitToPlace));
        }
        
        aiResources -= unitToPlace.cost;
        aiPlacementCooldown = 40 + Math.floor(Math.random() * 40); 
    } else {
        aiPlacementCooldown = 30;
    }
}

function handleGridClick() {
    if (isPaused || !gameActive) return;

    let gy = Math.floor((mouse.y - TOP_OFFSET) / CELL_SIZE) * CELL_SIZE + TOP_OFFSET;
    
    if (gameMode === 'attack') {
        if (!selectedUnit) return;
        if (mouse.y >= TOP_OFFSET && mouse.y <= TOP_OFFSET + (GRID_ROWS * CELL_SIZE)) {
            if (resourceAmount >= selectedUnit.cost) {
                enemies.push(new Enemy(gy, selectedUnit));
                resourceAmount -= selectedUnit.cost;
                selectedUnit = null;
                updateCardVisuals();
                sfx.place();
            } else {
                sfx.error();
            }
        } else {
            selectedUnit = null;
            updateCardVisuals();
        }
        return;
    }

    let gx = Math.floor((mouse.x - LEFT_OFFSET) / CELL_SIZE) * CELL_SIZE + LEFT_OFFSET;
    if (mouse.x < LEFT_OFFSET || mouse.x > LEFT_OFFSET + (GRID_COLS * CELL_SIZE) || mouse.y < TOP_OFFSET || mouse.y > TOP_OFFSET + (GRID_ROWS * CELL_SIZE)) {
        if (selectedUnit) {
            selectedUnit = null;
            updateCardVisuals();
        }
        return;
    }

    if (shovelActive) {
        const idx = defenders.findIndex(d => d.x === gx && d.y === gy);
        if (idx !== -1) { 
            defenders.splice(idx, 1); 
            addFloatyText("TRASHED!", gx + CELL_SIZE/2, gy); 
            toggleShovel(); 
            sfx.hit(); 
        }
        return;
    }

    if (!selectedUnit) return;
    let occupied = defenders.some(d => d.x === gx && d.y === gy);
    if (!occupied && resourceAmount >= selectedUnit.cost) {
        defenders.push(new Defender(gx, gy, selectedUnit));
        resourceAmount -= selectedUnit.cost; 
        selectedUnit = null; 
        updateCardVisuals();
        sfx.place();
    } else if (!occupied) {
        sfx.error();
    }
}

function animate() {
    ctx.clearRect(0, 0, LOGICAL_WIDTH, LOGICAL_HEIGHT);

    if (gameActive) {
        for (let cell of grid) cell.draw();

        if (!isPaused) {
            for(let s=0; s<gameSpeed; s++) {
                frames++;
                handleSpawns();
                if (gameMode === 'attack') handleAI();

                defenders.forEach(d => d.update());
                enemies.forEach(e => e.update());
                projectiles.forEach(p => p.update());
                resources.forEach(r => r.update());
                floatyTexts.forEach(f => f.update());
                
                for (let i = defenders.length - 1; i >= 0; i--) {
                    if (defenders[i].hp <= 0) {
                        if (defenders[i].type.trait === 'death_bomb') {
                            enemies.forEach(e => {
                                let dist = Math.hypot(e.x - (defenders[i].x + CELL_SIZE/2), e.y - (defenders[i].y + CELL_SIZE/2));
                                if (dist < CELL_SIZE * 2) e.takeDamage(1500); 
                            });
                            addFloatyText("KABOOM!", defenders[i].x + CELL_SIZE/2, defenders[i].y);
                            sfx.boom();
                        }
                        if (gameMode === 'attack' && Math.random() < 0.6) resources.push(new Resource(defenders[i].x, defenders[i].y));
                        defenders.splice(i, 1);
                    }
                }
                
                for (let i = enemies.length - 1; i >= 0; i--) {
                    if (enemies[i].hp <= 0) {
                        if (gameMode === 'defend') {
                            if (Math.random() < 0.25) resources.push(new Resource(enemies[i].x, enemies[i].y));
                            resourceAmount += 10; 
                        }
                        enemies.splice(i, 1);
                    }
                }
                for (let i = projectiles.length - 1; i >= 0; i--) if (projectiles[i].markedForDeletion) projectiles.splice(i, 1);
                for (let i = resources.length - 1; i >= 0; i--) if (resources[i].markedForDeletion) resources.splice(i, 1);
                for (let i = floatyTexts.length - 1; i >= 0; i--) if (floatyTexts[i].markedForDeletion) floatyTexts.splice(i, 1);

                if (gameMode === 'defend') {
                    if (waveStarted && enemiesToSpawn.length === 0 && enemies.length === 0 && !waveEnding) {
                        waveEnding = true;
                        setTimeout(() => { 
                            wave++; 
                            gameActive = false; 
                            document.getElementById('screens-bg').style.display = 'block'; 
                            setupSelectionScreen(); 
                        }, 1500); 
                    }
                } else if (gameMode === 'attack') {
                    let cheapestVeggie = Math.min(...userDeck.map(u => u.cost));
                    if (enemies.length === 0 && resourceAmount < cheapestVeggie) {
                        if (frames % 600 === 0) endGame(false);
                    }
                }
            }
        }

        // Draw everything even if paused
        defenders.forEach(d => d.draw());
        enemies.forEach(e => e.draw());
        projectiles.forEach(p => p.draw());
        resources.forEach(r => r.draw());
        floatyTexts.forEach(f => f.draw());

        if (isPaused) {
             ctx.save(); 
             ctx.fillStyle = 'rgba(0, 0, 0, 0.6)'; 
             ctx.fillRect(0, 0, LOGICAL_WIDTH, LOGICAL_HEIGHT);
             ctx.font = 'bold 80px ' + GAME_FONT; 
             ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
             ctx.fillStyle = '#ffcc00'; 
             ctx.fillText(`PAUSED`, LOGICAL_WIDTH/2, LOGICAL_HEIGHT/2); 
             ctx.restore();
        }

        if (waveEnding) {
             ctx.save(); ctx.fillStyle = 'rgba(0, 0, 0, 0.7)'; ctx.fillRect(0, LOGICAL_HEIGHT/2 - 60, LOGICAL_WIDTH, 120);
             ctx.font = 'bold 60px ' + GAME_FONT; ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
             ctx.fillStyle = '#ff00de'; ctx.fillText(`WAVE COMPLETE!`, LOGICAL_WIDTH/2, LOGICAL_HEIGHT/2); ctx.restore();
        }
    } else {
        // Draw the background grid just to keep the visual state consistent if transitioning
        for (let cell of grid) cell.draw();
    }
    
    requestAnimationFrame(animate);
}

function addFloatyText(text, x, y) { floatyTexts.push(new FloatyText(text, x, y)); }
